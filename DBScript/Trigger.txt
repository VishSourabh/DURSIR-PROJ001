//ID_CREATOR Trigger
create or replace TRIGGER ID_CREATOR 
BEFORE INSERT ON CUSTOMER_MASTER 
 FOR EACH ROW
BEGIN
  SELECT SEQ_ID.nextval
  INTO :new.CUST_ID
  FROM dual;
END;


//RECEIPT_ID Trigger
create or replace TRIGGER "SOURABH"."RECEIPT_ID" 
BEFORE INSERT ON RECEIPT_STATEMENT 
FOR EACH ROW
BEGIN
 SELECT SEQ_ID.nextval
  INTO :new.POLICY_NO
  FROM dual;
END;
